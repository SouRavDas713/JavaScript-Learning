let timeoutId; let intervalId; function startTimeout() { timeoutId = setTimeout(() => { document.getElementById('timeoutDiv').innerText = "Hello from Timeout!"; }, 3000); // 3 seconds delay } function stopTimeout() { clearTimeout(timeoutId); document.getElementById('timeoutDiv').innerText = "Timeout Stopped."; } function startInterval() { intervalId = setInterval(() => { const div = document.getElementById('intervalDiv'); const p = document.createElement('p'); p.innerText = "Hello from Interval!"; div.appendChild(p); }, 2000); // every 2 seconds } function stopInterval() { clearInterval(intervalId); const div = document.getElementById('intervalDiv'); const p = document.createElement('p'); p.innerText = "Interval Stopped."; div.appendChild(p); }
